                    <p class="text-[11px] text-gray-500 uppercase tracking-normal mt-2">สถิติพิกัด: ${hoursLived.toLocaleString()} ชั่วโมง / ${minutesLived.toLocaleString()} นาที แห่งการมีตัวตน</p>

                </div>

            `;

 

            generateGrid();

            document.getElementById('step-login').classList.add('hidden');

            document.getElementById('step-ritual').classList.remove('hidden');

            window.scrollTo(0,0);

        }

 

        function generateGrid() {

            const grid = document.getElementById('card-selection-grid');

            grid.innerHTML = '';

            for (let i = 0; i < 78; i++) {

                const card = document.createElement('div');

                card.className = 'card-selectable';

                card.innerHTML = `<span>TAROT</span>`;

                card.onclick = () => performScan(i);

                grid.appendChild(card);

            }

        }

 

        async function performScan(cardIndex) {

            const prayerText = document.getElementById('prayer').value.trim();

            if (!prayerText) {

                alert("กรุณาระบุถ้อยแถลงหรือปมในใจของท่านก่อนสัมผัสไพ่ เพื่อให้สัญชาตญาณเดิมนำทาง");

                return;

            }

            currentUser.prayer = prayerText;

 

            document.getElementById('step-ritual').classList.add('hidden');

            document.getElementById('step-loading').classList.remove('hidden');

            window.scrollTo(0,0);

 

            const resultCard = fullTarotDeck[Math.floor(Math.random() * fullTarotDeck.length)];

            const systemPrompt = `คุณคืออาจารย์ฟาณ (Ajarn Faan) กุนซือวิเคราะห์ดวงชะตา (Scorpio-Saturn class) 

            วินิจฉัยดวงชะตาด้วยมาตรวัดระดับโลก นิ่ง ตรง และทรงอำนาจ

            ใช้ "ภาษามนุษย์ปกติ" ที่สื่อสารกับคนทุกวัย (18-89 ปี) ได้อย่างลึกซึ้ง เข้าใจง่าย ไม่ลิเก ไม่กำกวม

            

            กฎแห่งสัมพัทธภาพของจิต: จงเชื่อมโยงปมในใจที่ระบุมา เข้ากับสัญลักษณ์ของหน้าไพ่ โดยใช้สัญชาตญาณเดิมนำทาง

            

            กติกาเข้มงวด:

            - ห้ามเอ่ยถึงชื่อวันเกิดเด็ดขาด (เช่น ห้ามพูดว่า "คนวันพฤหัสบดี") ระบบรู้พิกัดนี้อยู่แล้ว ให้ใช้วินิจฉัยจุดบอดเลย

            - ห้ามหยาบคาย ภาษาต้องดูแพง เป็นการวินิจฉัย (Diagnosis) แบบกุนซือระดับสูง

            - วิเคราะห์ 'ปมในใจ' (${currentUser.prayer}) ผ่านหน้าไพ่ ${resultCard.name}

            - นำ 'สถิติการใช้ชีวิต' (${currentUser.daysLived} วัน) มาคำนวณน้ำหนักบาดแผลและทางตัน

            - ห้ามใช้ศัพท์ก่อสร้าง

            

            เขียน 3-4 ย่อหน้าสั้นๆ ที่ทำให้ลูกดวงรู้สึกจุกและตื่นรู้จากภายใน`;

 

            // Setup Countdown and Fetch

            let secondsLeft = 10;

            const countdownEl = document.getElementById('scan-countdown');

            countdownEl.innerText = secondsLeft;

 

            let aiResponse = "";

            let fetchDone = false;

 

            const timerInterval = setInterval(() => {

                secondsLeft--;

                if (secondsLeft >= 0) countdownEl.innerText = secondsLeft;

                if (secondsLeft <= 0) {

                    clearInterval(timerInterval);

                    checkFinalTransition();

                }

            }, 1000);

 

            function checkFinalTransition() {

                if (fetchDone && secondsLeft <= 0) displayResult(aiResponse, resultCard);

            }

 

            try {

                const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({ contents: [{ parts: [{ text: systemPrompt }] }] })

                });

                const data = await res.json();

                aiResponse = data.candidates?.[0]?.content?.parts?.[0]?.text || "พิกัดสมาธิขัดข้อง";

                fetchDone = true;

                checkFinalTransition();

            } catch (e) {

                aiResponse = "การเชื่อมต่อสมาธิถูกรบกวนเนื่องจากสัญญาณรบกวนภายนอก กรุณาลองใหม่";

                fetchDone = true;

                checkFinalTransition();

            }

        }

 

        function displayResult(text, card) {

            document.getElementById('step-loading').classList.add('hidden');

            document.getElementById('step-result').classList.remove('hidden');

            window.scrollTo(0,0);

            

            document.getElementById('final-card-container').innerHTML = `

                <div class="tarot-card-result reveal-scale bg-white">

                    <img src="${card.img}" class="card-face-img" alt="${card.name}">

                    <div class="absolute bottom-0 left-0 right-0 p-5 bg-black/80 backdrop-blur-md text-center border-t border-gold-foil/30">

                        <div class="font-cinzel text-gold-foil text-sm font-bold uppercase tracking-normal">${card.name}</div>

                    </div>

                </div>`;

            

            document.getElementById('card-name').innerText = card.name;

            document.getElementById('card-subtitle').innerText = card.sub;

            document.getElementById('card-description').innerHTML = text.split('\n').filter(p=>p.trim()).map(p=>`<p class="mb-4">${p}</p>`).join('');

        }
